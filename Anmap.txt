#include<iostream>
using namespace std;

struct canh
{
	int Dinh1;
	int Dinh2;
};
typedef struct canh CANH;


int main()
{
	int n;
	cout << "Nhap So Dinh: \n";
	cin >> n;
	int m;
	cout << "Nhap So Canh: \n";
	cin >> m;
	CANH E[100];
	for (int i = 0; i < m; i++)
	{
		cout << "\nNhap Canh "<<i+1<<": \n" << "Dinh 1: ";
		cin >> E[i].Dinh1;
		cout << "Dinh 2: ";
		cin >> E[i].Dinh2;
	}
	int MangThu[100][100] = { 0 };
	for (int i = 0; i < n; i++)
	{
		for (int j = 0; j < n; j++)
		{
			for (int k = 0; k < m; k++)
			{
				if (E[k].Dinh1 - 1 == i && E[k].Dinh2 - 1 == j)
					MangThu[i][j]++;
			}

		}
	}
	cout << "\n=====================\n";
	for (int i = 0; i < n; i++)
	{
		for (int j = 0; j < n; j++)
		{
			cout << MangThu[i][j] << " ";
		}
		cout << endl;
	}
		
	system("pause");
	return 0;
}
